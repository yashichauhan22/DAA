#include<iostream>
using namespace std;
void binarysearch(int a[],int n,int key)
{
    int l=0,h=n-1,m,i,pos,f=0,c=0;
    while(l<=h)
    {
        m=(l+h)/2;
        c++;
        if(a[m]==key)
        {
            f=1;
            break;
        }
        else if(a[m]>key)
        {
            h=m-1;
        }
        else if(a[m]<key)
        {
            l=m+1;
        }
    }
    if(f==1)
    {
        cout<<"present "<<c<<endl;
    }
    else
    {
        cout<<"not present "<<endl;
    }
}
int main()
{
    int m,j;
    cout<<"enter number";
    cin>>m;
    for(j=0;j<m;j++)
    {
	int n,key;
	cout<<"enter size";
	cin>>n;
	cout<<"enter key";
	cin>>key;
	int i,a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	binarysearch(a,n,key);
    }
	return 0;
}
